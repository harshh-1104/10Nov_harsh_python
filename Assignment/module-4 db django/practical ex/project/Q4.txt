# Q4: Python Program to create a Django project and understand its directory structure

# To create a new Django project via the command line, you use:
# django-admin startproject project_name

import subprocess
import os

def create_django_project(name="my_test_project"):
    """
    This function demonstrates how to create a Django project
    using Python's subprocess module.
    """
    print(f"--- 🚀 Starting Project Creation: {name} ---")
    try:
        # The command to create a Django project
        result = subprocess.run(["django-admin", "startproject", name], capture_output=True, text=True)
        
        if result.returncode == 0:
            print(f"✅ Success! Django project '{name}' has been created.")
            explain_structure(name)
        else:
            print(f"❌ Error while creating project: {result.stderr}")
            
    except FileNotFoundError:
        print("❌ 'django-admin' command not found. Please install Django first: pip install django")
    except Exception as e:
        print(f"❌ An error occurred: {e}")

def explain_structure(name):
    """
    Provides a detailed explanation of the Django project's directory structure.
    """
    print("\n--- 📂 Django Project Directory Structure ---")
    structure = f"""
{name}/                          (Root Folder)
├── manage.py                   (Interactive command-line utility for managing the project)
└── {name}/                      (Internal package containing configuration files)
    ├── __init__.py             (Empty file marking this folder as a Python package)
    ├── settings.py             (Storage for all project-wide configuration and settings)
    ├── urls.py                 (The "Table of Contents" mapping URLs to views)
    ├── asgi.py                 (Entry-point for ASGI-compatible web servers)
    └── wsgi.py                 (Entry-point for WSGI-compatible web servers)
    """
    print(structure)
    print("--- 📄 File Details ---")
    print("1. manage.py: Used to run development server, manage database migrations, and more.")
    print("2. settings.py: Defines database connection, static file paths, installed apps, etc.")
    print("3. urls.py: Defines how different URL patterns should be handled.")
    print("4. wsgi.py / asgi.py: Used when deploying your site to production servers.")

if __name__ == "__main__":
    create_django_project("my_example_project")
